## `MainActivity.kt` (Android 15+, Kotlin, single scrollable screen)

```kotlin
package com.sel2in.suzysnooze

import android.Manifest
import android.content.Intent
import android.net.Uri
import android.os.Build
import android.os.Bundle
import android.os.PowerManager
import android.provider.Settings
import android.widget.*
import androidx.activity.result.contract.ActivityResultContracts
import androidx.appcompat.app.AppCompatActivity
import androidx.core.content.ContextCompat
import androidx.work.*

import java.util.concurrent.TimeUnit

class MainActivity : AppCompatActivity() {

    private lateinit var hourPicker: NumberPicker
    private lateinit var minutePicker: NumberPicker

    private lateinit var chkWifi: CheckBox
    private lateinit var chkAirplane: CheckBox
    private lateinit var chkInternetPanel: CheckBox
    private lateinit var chkBatteryIgnore: CheckBox
    private lateinit var chkMute: CheckBox

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        setupQuickButtons()
        setupPickers()
        setupCheckboxes()
        setupPermissionButtons()
        setupStartCustom()
    }

    private fun setupQuickButtons() {
        val map = mapOf(
            R.id.btn1 to 1,
            R.id.btn2 to 2,
            R.id.btn5 to 5,
            R.id.btn20 to 20,
            R.id.btn30 to 30
        )

        map.forEach { (id, minutes) ->
            findViewById<Button>(id).setOnClickListener {
                startSnooze(minutes.toLong())
            }
        }
    }

    private fun setupPickers() {
        hourPicker = findViewById(R.id.hourPicker)
        minutePicker = findViewById(R.id.minutePicker)

        hourPicker.minValue = 1
        hourPicker.maxValue = 15

        minutePicker.minValue = 0
        minutePicker.maxValue = 59
    }

    private fun setupCheckboxes() {
        chkWifi = findViewById(R.id.chkWifi)
        chkAirplane = findViewById(R.id.chkAirplane)
        chkInternetPanel = findViewById(R.id.chkInternetPanel)
        chkBatteryIgnore = findViewById(R.id.chkBatteryIgnore)
        chkMute = findViewById(R.id.chkMute)
    }

    private fun setupPermissionButtons() {

        findViewById<Button>(R.id.btnBatteryOpt).setOnClickListener {
            val intent = Intent(
                Settings.ACTION_REQUEST_IGNORE_BATTERY_OPTIMIZATIONS,
                Uri.parse("package:$packageName")
            )
            startActivity(intent)
        }

        findViewById<Button>(R.id.btnInternetPanel).setOnClickListener {
            startActivity(Intent(Settings.Panel.ACTION_INTERNET_CONNECTIVITY))
        }

        findViewById<Button>(R.id.btnAirplaneSettings).setOnClickListener {
            startActivity(Intent(Settings.ACTION_AIRPLANE_MODE_SETTINGS))
        }
    }

    private fun setupStartCustom() {
        findViewById<Button>(R.id.btnStartCustom).setOnClickListener {
            val totalMinutes =
                hourPicker.value * 60L + minutePicker.value
            startSnooze(totalMinutes)
        }
    }

    private fun startSnooze(minutes: Long) {

        if (chkMute.isChecked) {
            VolumeManager(this).muteAndSave()
        }

        if (chkInternetPanel.isChecked) {
            startActivity(Intent(Settings.Panel.ACTION_INTERNET_CONNECTIVITY))
        }

        if (chkAirplane.isChecked) {
            startActivity(Intent(Settings.ACTION_AIRPLANE_MODE_SETTINGS))
        }

        val delayMillis = minutes * 60 * 1000

        val work = OneTimeWorkRequestBuilder<RestoreWorker>()
            .setInitialDelay(delayMillis, TimeUnit.MILLISECONDS)
            .build()

        WorkManager.getInstance(this)
            .enqueueUniqueWork(
                "restore_volume",
                ExistingWorkPolicy.REPLACE,
                work
            )

        Toast.makeText(this, "Snooze started for $minutes minutes", Toast.LENGTH_SHORT).show()
    }
}
```

Goal: DND ON, only calls + notifications muted (incl. vibration), alarms + media untouched, then restore to normal ring + DND OFF after X minutes. Works on Android 14–15+ (OnePlus Nord CE, Samsung). Includes fallback.

CRITICAL: DND Policy MUST explicitly allow ALARMS and MEDIA categories to prevent Android from silencing media playback. Setting policy to 0 (block all) causes DND to silence media on some devices.

1) Manifest
<manifest ...> <uses-permission android:name="android.permission.ACCESS_NOTIFICATION_POLICY"/> <uses-permission android:name="android.permission.MODIFY_AUDIO_SETTINGS"/> <uses-permission android:name="android.permission.FOREGROUND_SERVICE"/> <uses-permission android:name="android.permission.RECEIVE_BOOT_COMPLETED"/> <application ...> <service android:name=".DndAccessibilityService" android:permission="android.permission.BIND_ACCESSIBILITY_SERVICE" android:exported="false"> <intent-filter> <action android:name="android.accessibilityservice.AccessibilityService"/> </intent-filter> <meta-data android:name="android.accessibilityservice" android:resource="@xml/accessibility_config"/> </service> <receiver android:name=".BootReceiver" android:exported="false"> <intent-filter> <action android:name="android.intent.action.BOOT_COMPLETED"/> </intent-filter> </receiver> </application> </manifest>

NotificationManager nm = (NotificationManager) getSystemService(Context.NOTIFICATION_SERVICE); if (!nm.isNotificationPolicyAccessGranted()) { startActivity(new Intent(Settings.ACTION_NOTIFICATION_POLICY_ACCESS_SETTINGS)); } // battery optimization off startActivity(new Intent(Settings.ACTION_IGNORE_BATTERY_OPTIMIZATION_SETTINGS)); // optional fallback startActivity(new Intent(Settings.ACTION_ACCESSIBILITY_SETTINGS)); 
3) State model
class AudioState { int ringerMode; int ringVolume; int notifVolume; int alarmVolume; }

DND ON
Calls + notifications silent
No vibration
Alarms/media unchanged - NEVER touch STREAM_ALARM or STREAM_MUSIC volumes
public void enableFocusMode(Context ctx, int minutes) { NotificationManager nm = (NotificationManager) ctx.getSystemService(Context.NOTIFICATION_SERVICE); AudioManager am = (AudioManager) ctx.getSystemService(Context.AUDIO_SERVICE); // ---- Save state ---- AudioState s = new AudioState(); s.ringerMode = am.getRingerMode(); s.ringVolume = am.getStreamVolume(AudioManager.STREAM_RING); s.notifVolume = am.getStreamVolume(AudioManager.STREAM_NOTIFICATION); s.alarmVolume = am.getStreamVolume(AudioManager.STREAM_ALARM); saveState(ctx, s); // ---- DND ON (block calls + notifications, ALLOW alarms + media) ---- nm.setInterruptionFilter(NotificationManager.INTERRUPTION_FILTER_PRIORITY); // CRITICAL: Policy MUST allow ALARMS and MEDIA to prevent DND from silencing media NotificationManager.Policy policy = new NotificationManager.Policy( NotificationManager.Policy.PRIORITY_CATEGORY_ALARMS | NotificationManager.Policy.PRIORITY_CATEGORY_MEDIA, // Allow alarms + media 0, // no priority callers 0 // no priority messages ); nm.setNotificationPolicy(policy); // ---- Force NO vibration + silent for calls/notifications ---- am.setRingerMode(AudioManager.RINGER_MODE_SILENT); am.setStreamVolume(AudioManager.STREAM_RING, 0, 0); am.setStreamVolume(AudioManager.STREAM_NOTIFICATION, 0, 0); // DO NOT TOUCH - these must remain at current volume: // STREAM_ALARM // STREAM_MUSIC // ---- schedule restore ---- scheduleRestore(ctx, minutes); } 
5) Restore (FIXED - no vibrate issue)
public class RestoreWorker extends Worker { public RestoreWorker(Context ctx, WorkerParameters params) { super(ctx, params); } @Override public Result doWork() { Context ctx = getApplicationContext(); NotificationManager nm = (NotificationManager) ctx.getSystemService(Context.NOTIFICATION_SERVICE); AudioManager am = (AudioManager) ctx.getSystemService(Context.AUDIO_SERVICE); AudioState s = loadState(ctx); // ---- DND OFF ---- nm.setInterruptionFilter(NotificationManager.INTERRUPTION_FILTER_ALL); // ---- FORCE NORMAL (critical) ---- am.setRingerMode(AudioManager.RINGER_MODE_NORMAL); // ---- restore volumes safely ---- int ringVol = Math.max(s.ringVolume, 3); int notifVol = Math.max(s.notifVolume, 3); am.setStreamVolume(AudioManager.STREAM_RING, ringVol, 0); am.setStreamVolume(AudioManager.STREAM_NOTIFICATION, notifVol, 0); am.setStreamVolume(AudioManager.STREAM_ALARM, s.alarmVolume, 0); return Result.success(); } }
